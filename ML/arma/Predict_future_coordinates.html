def mean_std_agg(df):
    
    ops_m = ['xm','ym','zm']
    ops = ['x','y','z']
    for i,op in enumerate(ops_m):
        # data['column'].rolling(5).mean()
        df[op] = df[ops[i]].rolling(5).mean()
    
    ops_std = ['x_std','y_std','z_std']
    for k,op in enumerate(ops_std):
        df[op] = df[ops[k]].rolling(5).mean()
        
    df = df.iloc[4:]
    return df
